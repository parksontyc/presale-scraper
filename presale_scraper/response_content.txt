	var ctrlKeyDown = false;
	//開啟新窗
	function open_window(url,category,event,desc){ 
		window.open(url);
		trackEvent(category,event,desc);
	}
	function keydown(e) { 

		if ((e.which || e.keyCode) == 116 || ((e.which || e.keyCode) == 82 && ctrlKeyDown)) {
			// Pressing F5 or Ctrl+R
			e.preventDefault();
		} else if ((e.which || e.keyCode) == 17) {
			// Pressing  only Ctrl
			ctrlKeyDown = true;
		}
	};

	function keyup(e){
		// Key up Ctrl
		if ((e.which || e.keyCode) == 17) 
			ctrlKeyDown = false;
	};
